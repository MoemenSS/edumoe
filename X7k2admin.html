Required Supabase Setup SQL

Run this in Supabase → SQL Editor to enable IP logging and admin access:

-- 1. Create admin_logs table for IP logging
CREATE TABLE IF NOT EXISTS admin_logs (
  id         BIGSERIAL PRIMARY KEY,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  email      TEXT,
  ip_address TEXT,
  user_agent TEXT,
  success    BOOLEAN,
  note       TEXT
);

-- 2. Enable RLS but allow inserts from all (so failed logins log too)
ALTER TABLE admin_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow public insert" ON admin_logs
  FOR INSERT TO anon, authenticated WITH CHECK (true);

CREATE POLICY "Admins can read logs" ON admin_logs
  FOR SELECT TO authenticated USING (true);

-- 3. Add is_admin column to profiles (if not exists)
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS is_admin BOOLEAN DEFAULT FALSE;

-- 4. Mark your own account as admin (replace with your user ID)
-- UPDATE profiles SET is_admin = TRUE WHERE id = 'YOUR-USER-UUID';
